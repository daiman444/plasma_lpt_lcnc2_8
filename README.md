# Plasma_LPT
Plasma config for LinuxCNC

# HotKeys.

"r" - run program;\
"t" - pause program;\
"y" - resume program;\
"esc" - abort program;\
"u" - jog speed continuous;\
"i" - jog speed increment;

"q" - arc ok;\
"a" - thc up"\
"z" - thc down;

"s" - motion.adaptive-feed +1;\
"x" - motion.adaptive-feed -1;

"d" - volts +1;\
"c" - volts -1;
